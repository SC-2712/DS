#include <stdio.h>

int main() {
    int n, i, search, found = 0;

    printf("Enter number of students: ");
    scanf("%d", &n);

    int roll[n];

    printf("Enter roll numbers of students (in random order):\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &roll[i]);
    }

    printf("Enter roll number to search: ");
    scanf("%d", &search);

    for(i = 0; i < n; i++) {
        if(roll[i] == search) {
            found = 1;
            break;
        }
    }

    if(found)
        printf("Student with roll number %d attended the training program.\n", search);
    else
        printf("Student with roll number %d did NOT attend the training program.\n", search);

    return 0;
}
