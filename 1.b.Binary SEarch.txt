#include <stdio.h>

// Function to perform binary search
int binarySearch(int arr[], int n, int key) {
    int low = 0, high = n - 1;

    while (low <= high) {
        int mid = (low + high) / 2;

        if (arr[mid] == key)
            return mid;  // found
        else if (arr[mid] < key)
            low = mid + 1;
        else
            high = mid - 1;
    }

    return -1; // not found
}

int main() {
    int roll[100];
    int n, choice, key, result;

    printf("Enter number of students who attended the training program: ");
    scanf("%d", &n);

    printf("\nEnter roll numbers in sorted order:\n");
    for (int i = 0; i < n; i++) {
        printf("Roll[%d]: ", i + 1);
        scanf("%d", &roll[i]);
    }

    while (1) {
        printf("\n===== MENU =====\n");
        printf("1. Display Roll Numbers\n");
        printf("2. Search a Student (Binary Search)\n");
        printf("3. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("\nStudents who attended training:\n");
                for (int i = 0; i < n; i++)
                    printf("%d ", roll[i]);
                printf("\n");
                break;

            case 2:
                printf("\nEnter roll number to search: ");
                scanf("%d", &key);
                result = binarySearch(roll, n, key);

                if (result != -1)
                    printf(" Student with roll number %d attended the training (found at position %d).\n", key, result + 1);
                else
                    printf("Student with roll number %d did NOT attend the training.\n", key);
                break;

            case 3:
                printf("\nExiting program. Goodbye!\n");
                return 0;

            default:
                printf("\nInvalid choice! Please try again.\n");
        }
    }
}
